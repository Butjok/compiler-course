// C functions
extern int time(int x);
extern void srand(int x);
extern int rand();

extern int sleep(int x);
extern int usleep(int x);

// C extensions
extern float frand();


int main()
{
	srand(time(0));

	int total = 0;
	int in = 0;

	for (int i = 0; i < 10000000; i++;)
	{
		float x = frand();
		float y = frand();

		if (x*x + y*y <= 1)
		{
			in++;
		}

		total++;
	}

	float pi = 4.0 * in / total;

	print("Calculated PI = ", pi, "\n");
	print("Predefined PI = ", pi(), "\n");

	return 0;
}
